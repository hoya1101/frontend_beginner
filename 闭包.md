### 使用闭包

#### 私有变量

```javascript
function Ninja() {
  var feints = 0;
  this.getFeints = function() {
    return feints;
  }
}

var ninja = new Ninja();
console.log(ninja.getFeints(), feints);
```

#### 回调和计时器

```html
<div id="testSubject"></div>

<button type="button" id="testButton">GO!</div>

<script type="text/javascript">
  $('#testButton').click(function() {
    var elem$ = $('testSubject');
    
    elem$.html('Loading...');
    
    $.ajax({
      url : 'test.html',
      success : function(html) {
        elem$.html(html);
      }
    });
  });
</script>
```

计数器
```html
<div id="box">box</div>
<script type="text/javascript">
  function animateIt(elementId) {
    //debugger
    var elem = document.getElementById(elementId),
        tick = 0,
        timer = setInterval(function(){
          if (tick < 100) {
            elem.style.left = elem.style.top = tick + "px";
            console.log( elem.style.left);
            tick++;
          } else {
            clearInterval(timer);
            console.log(tick, elem, timer)
          }
        }, 10)
  }
  
  animateIt('box');
</script>
```

## 绑定函数上下文

```javascript
<button id="test">Click me!</button>

var button = {
  clicked: false,
  click : function() {
    this.clicked = true;
    console.log(clicked);
  }
}
var elem = document.getElementById('test');
elem.addEventListener('click', button.click, false);
```
调用函数`button.click`,该函数的上下文就是button. 但是浏览器事件处理是同认为函数调用的上下文是事件的目标对象，导致其上下文是`<button>`元素。

```javascript
function bind(context, name) {
  return function() {
    return context[name].apply(context, arguments);
  }
}
```

## 偏应用函数

"分部应用"，在函数调用之前，可以预先传入一些函数，它返回一个含有预处理参数的新函数，以便后期调用。

```javascript
String.prototype.csv = String.prototype.split.partial(/,\s*/)

var result = ('adad,    fff,   a').csv();

Function.prototype.curry = function() {
  var fn = this,
      args = Array.prototype.slice.call(arguments);
  return function() {
    return fn.apply(this, args.concat(Array.prototype.slice.call(arguments)));
  };
}
```
